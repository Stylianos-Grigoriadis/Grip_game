import matplotlib.pyplot as plt
import numpy as np

def sine(Number_of_targets,frequency, Total_Time):
    x = np.arange(0, Number_of_targets)
    y = np.sin(x*frequency)
    y = y - np.min(y)
    y = y*100/np.max(y)
    time = np.arange(0, Total_Time, Total_Time/Number_of_targets)
    plt.plot(time,y)
    plt.show()
    return time,y

sine(100,1,60)
